### インターフェース設計

```solidity
// GovernanceSwitch.sol
interface IGovernanceSwitch {
    enum GovernanceMode { 
        CENTRALIZED,    // OFF: 単一管理者
        MULTISIG,       // OFF: マルチシグ
        DECENTRALIZED   // ON: DAO + Security Council
    }
    
    function setGovernanceMode(GovernanceMode mode) external;
    function getGovernanceMode() external view returns (GovernanceMode);
}

// TokenSwitch.sol
interface ITokenSwitch {
    enum TokenMode {
        DISABLED,       // OFF: トークンなし
        BASIC,          // ON: 基本トークン
        FULL            // ON: veQS + Staking + Rewards
    }
    
    function setTokenMode(TokenMode mode) external;
    function getTokenMode() external view returns (TokenMode);
}
```
